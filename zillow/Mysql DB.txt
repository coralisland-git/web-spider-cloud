CREATE TABLE parcel_estimate (
id int auto_increment not null primary key,
folio varchar(10),
pin varchar(50),
owner varchar(100),
site_addr varchar(50) ,
site_city varchar(30),
site_state varchar(20),
site_zip varchar(20),
bed int(11),
bath int(11),
square_footage int(11),
zestimate int(20),
mark varchar(20),
);
